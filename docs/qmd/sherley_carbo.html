dat <- readRDS("sherley_final_master.rds") %>%

# Create eGFR ratio: CKD-EPI Cr-Cys (unindexed) / Cockcroft-Gault
# Also create per-10 mL/min rescaled versions of cockcroft and CKD-EPI for
# reporting (HRs per 1 mL/min are clinically uninterpretable; convention in
# nephro-onc is per 10 mL/min — bio_agent confirmed). Adjusters keep raw scale.
dat <- dat %>%
  mutate(
    # Capped versions (for dose discrepancy calculations)
    cockcroft_cap125 = pmin(cockcroft, 125),
    # ckd_epi_gfr_cre_cys_unindex_cap125 already exists in data
    # Ratio uses UNCAPPED eGFR
    egfr_ratio    = ckd_epi_gfr_cre_cys_unindex / cockcroft,
    egfr_ratio_per10 = -egfr_ratio * 10,  # per 0.1-unit *decrease* reporting scale
    egfr_discrepancy = ckd_epi_gfr_cre_cys_unindex - cockcroft,
    # Dose discrepancy uses CAPPED eGFR (already computed in sherley_data.qmd)
    dose_discrep_per25increase = dose_discrepancy / 25,
    abs_dose_discrep = abs(dose_discrepancy),
    abs_dose_discrep_cat = cut(abs_dose_discrep,
      breaks = c(0, 25, 50, 75, 100, Inf),
      labels = c("<25", "25-49.99", "50-74.99", "75-99.99", ">=100"),
      right = FALSE, include.lowest = TRUE),
    # Binary exposure for adjusted CIF curves (Section 14): 1 if patient was
    # over-dosed by 25 mg or more (actual - predicted >= 25), else 0.
    dose_discrep_ge25 = as.integer(dose_discrepancy >= 25),
    # Signed dose discrepancy = actual - predicted; 10 bins symmetric around 0.
    signed_dose_discrep_cat = cut(dose_discrepancy,
      breaks = c(-Inf, -100, -75, -50, -25, 0, 25, 50, 75, 100, Inf),
      labels = c("<=-100", "-99.99 to -75", "-74.99 to -50",
                 "-49.99 to -25", "-24.99 to 0",
                 "0 to 24.99", "25 to 49.99", "50 to 74.99",
                 "75 to 99.99", ">=100"),
      right = FALSE, include.lowest = TRUE),
    cockcroft_per10 = cockcroft / 10,
    ckd_epi_per10   = ckd_epi_gfr_cre_cys_unindex / 10
  )

Event Counts by Outcome (Diagnostic)

For each outcome the table below shows the cohort N, events (status = 1 within 90 days), competing deaths (status = 2; not applicable to 90-day death which is modelled by Cox PH), censored, event rate, and events-per-variable (EPV) using the 4-variable adjuster set for this cohort (age, sex, score (Charlson), ici). As a rule of thumb, Fine-Gray fits need ≥ 5 × adj events to be stable; < 2 × adj almost always produces a singular design matrix, which is why the helper functions in this file fall back to a “Model could not be fit” placeholder where you see one.

.outcomes_tbl <- function(dat, cohort_label, n_adj = 4) {
  sumrow <- function(label, status_var, is_cox = FALSE) {
    s <- suppressWarnings(as.integer(as.character(dat[[status_var]])))
    n_total     <- sum(!is.na(s))
    n_event     <- sum(s == 1, na.rm = TRUE)
    n_compdeath <- if (is_cox) NA_integer_ else sum(s == 2, na.rm = TRUE)
    n_cens      <- sum(s == 0, na.rm = TRUE)
    epv         <- if (n_total > 0) n_event / n_adj else NA_real_
    fit_likely  <- if (is.na(epv))                "—"
                   else if (n_event >= 5 * n_adj) "✓ likely OK"
                   else if (n_event >= 2 * n_adj) "marginal"
                   else                           "✗ singular likely"
    tibble::tibble(
      Outcome            = label,
      N                  = n_total,
      Events             = n_event,
      `Competing deaths` = n_compdeath,
      Censored           = n_cens,
      `Event rate`       = if (n_total > 0) sprintf("%.1f%%", 100 * n_event / n_total) else "—",
      EPV                = if (!is.na(epv)) sprintf("%.1f", epv) else "—",
      `Fit reliable?`    = fit_likely
    )
  }
  dplyr::bind_rows(
    sumrow("Anemia Grade 2",                   "cmp_status_anemia_2"),
    sumrow("Anemia Grade 3",                   "cmp_status_anemia_3"),
    sumrow("Thrombocytopenia Grade 2",         "cmp_status_throm_2"),
    sumrow("Thrombocytopenia Grade 3",         "cmp_status_throm_3"),
    sumrow("AKI Grade 2",                      "cmp_status_aki_2"),
    sumrow("AKI (KDIGO)",                      "cmp_status_aki_kdigo"),
    sumrow("Composite Grade 2 AEs",            "cmp_status_grade_2_aes"),
    sumrow("Composite Grade 3 AEs",            "cmp_status_grade_3_aes"),
    sumrow("Overall Hospitalization",          "cmp_status_any_hosp"),
    sumrow("Platinum-Related Hospitalization", "cmp_status_side_effect"),
    sumrow("90-Day Death (Cox PH)",            "death_90", is_cox = TRUE)
  )
}

.adj_set <- function(pre_labs, egfr_adj) {
  # Minimal adjuster set for the Sherley sub-cohort (N=169): only the four
  # demographic/clinical confounders the user specified. The egfr_adj and
  # pre_labs arguments are accepted for backward compatibility but ignored.
  c("age", "sex", "score", "ici")
}

# fg_model: returns univariable + multivariable merged Fine-Gray table.
# If docx_name is supplied, also saves the table as a .docx in docx_dir.
fg_model <- function(time_var, status_var, pred_var, pre_labs = NULL,
                     egfr_adj = c("none", "cockcroft", "ckd_epi"),
                     docx_name = NULL) {
  egfr_adj <- match.arg(egfr_adj)
  adj <- .adj_set(pre_labs, egfr_adj)
  all_vars <- c(pred_var, adj)

  fmla_mv <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(all_vars, collapse = " + ")
  ))
  tbl_mv <- tryCatch(
    tidycmprsk::crr(fmla_mv, failcode = 1, data = dat) %>%
      tbl_regression(exp = TRUE,
                     pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
      add_n(),
    error = function(e) NULL
  )

  y_call <- str2lang(sprintf("Surv(%s, %s)", time_var, status_var))
  tbl_uv <- tryCatch(
    rlang::inject(
      tbl_uvregression(
        data = dat,
        method = tidycmprsk::crr,
        y = !!y_call,
        include = dplyr::all_of(all_vars),
        exponentiate = TRUE,
        pvalue_fun = function(x) style_pvalue(x, digits = 3),
        method.args = list(failcode = 1)
      )
    ),
    error = function(e) NULL
  )

  if (is.null(tbl_uv) && is.null(tbl_mv)) {
    return(knitr::asis_output(
      "> *Model could not be fit — Fine-Gray design matrix is singular (insufficient events or collinear covariates for this sub-cohort).*"
    ))
  }

  out <- if (is.null(tbl_mv)) {
    tbl_uv
  } else if (is.null(tbl_uv)) {
    tbl_mv
  } else {
    tbl_merge(
      tbls = list(tbl_uv, tbl_mv),
      tab_spanner = c("**Univariable**", "**Multivariable**")
    )
  }

  if (!is.null(docx_name)) save_tbl_docx(out, docx_name)
  out
}

cox_model <- function(time_var, status_var, pred_var, pre_labs = NULL,
                      egfr_adj = c("none", "cockcroft", "ckd_epi"),
                      docx_name = NULL) {
  egfr_adj <- match.arg(egfr_adj)
  adj <- .adj_set(pre_labs, egfr_adj)
  all_vars <- c(pred_var, adj)

  fmla_mv <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(all_vars, collapse = " + ")
  ))
  tbl_mv <- survival::coxph(fmla_mv, data = dat) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()

  y_call <- str2lang(sprintf("Surv(%s, %s)", time_var, status_var))
  tbl_uv <- rlang::inject(
    tbl_uvregression(
      data = dat,
      method = survival::coxph,
      y = !!y_call,
      include = dplyr::all_of(all_vars),
      exponentiate = TRUE,
      pvalue_fun = function(x) style_pvalue(x, digits = 3)
    )
  )

  out <- tbl_merge(
    tbls = list(tbl_uv, tbl_mv),
    tab_spanner = c("**Univariable**", "**Multivariable**")
  )

  if (!is.null(docx_name)) save_tbl_docx(out, docx_name)
  out
}

csh_model <- function(time_var, status_var, pred_var, pre_labs = NULL) {
  adj <- c("age", "sex", "score", "ici")
  d <- dat
  d$csh_status <- ifelse(d[[status_var]] == 1, 1L, 0L)
  fmla <- as.formula(paste0(
    "Surv(", time_var, ", csh_status) ~ ",
    paste(c(pred_var, adj), collapse = " + ")
  ))
  tryCatch(
    survival::coxph(fmla, data = d) %>%
      tbl_regression(exp = TRUE,
                     pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
      add_n(),
    error = function(e) .tbl_failed()
  )
}

NoteWhy anchor at the median (subHR = 1 at the reference)

Three reasons to anchor at the median:

  1. Interpretability: “A patient with eGFR ratio of 0.8 has ~1.5× the subHR compared to a typical patient (at the median).”
  2. Stability: The median sits in the densest part of the data, so estimates near it have the tightest confidence intervals. The CIs fan out as you move away — you can see this in the plots.
  3. No arbitrary choice: If we picked 1.0 arbitrarily, the plot would look almost identical here (because median ≈ 1), but in cohorts where the median is 0.9 or 1.1, picking 1.0 would either shift or distort the curve without changing the actual model.

plot_subhr_linear_ratio <- function(time_var, status_var, extra_adj = NULL,
                                    outcome_label) {
  base_adj <- c("age", "sex", "score", "ici")
  keep <- c("egfr_ratio", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(egfr_ratio = d$egfr_ratio, X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- tryCatch(
    cmprsk::crr(ftime = time_num, fstatus = stat_num,
                cov1 = X, failcode = 1),
    error = function(e) NULL
  )
  if (is.null(fit)) return(.plot_failed(outcome_label))

  beta   <- fit$coef[1]
  V_beta <- fit$var[1, 1]

  ref_x  <- median(d$egfr_ratio, na.rm = TRUE)
  grid_x <- seq(quantile(d$egfr_ratio, 0.025, na.rm = TRUE),
                quantile(d$egfr_ratio, 0.975, na.rm = TRUE),
                length.out = 100)

  log_hr <- beta * (grid_x - ref_x)
  se     <- sqrt((grid_x - ref_x)^2 * V_beta)

  df_plot <- tibble(
    x     = grid_x,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  ggplot(df_plot, aes(x = x, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "darkorange") +
    geom_line(color = "darkorange", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_x, linetype = "dotted", color = "grey40") +
    labs(x = "CKD-EPI Cr-Cys / CG ratio",
         y = "subHR (95% CI)",
         title = outcome_label) +
    coord_cartesian(xlim = c(0.65, 1.45), ylim = c(0, 4)) +
    theme_minimal(base_size = 11)
}

plot_subhr_linear <- function(time_var, status_var, extra_adj = NULL,
                              outcome_label) {
  base_adj <- c("age", "sex", "score", "ici")
  keep <- c("dose_discrepancy", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(dose_discrepancy = d$dose_discrepancy, X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- tryCatch(
    cmprsk::crr(ftime = time_num, fstatus = stat_num,
                cov1 = X, failcode = 1),
    error = function(e) NULL
  )
  if (is.null(fit)) return(.plot_failed(outcome_label))

  beta   <- fit$coef[1]
  V_beta <- fit$var[1, 1]

  ref_x  <- median(d$dose_discrepancy, na.rm = TRUE)
  grid_x <- seq(quantile(d$dose_discrepancy, 0.025, na.rm = TRUE),
                quantile(d$dose_discrepancy, 0.975, na.rm = TRUE),
                length.out = 100)

  log_hr <- beta * (grid_x - ref_x)
  se     <- sqrt((grid_x - ref_x)^2 * V_beta)

  df_plot <- tibble(
    x     = grid_x,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  ggplot(df_plot, aes(x = x, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "darkorange") +
    geom_line(color = "darkorange", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_x, linetype = "dotted", color = "grey40") +
    labs(x = "Dose Discrepancy (mg)",
         y = "subHR (95% CI)",
         title = outcome_label) +
    coord_cartesian(ylim = c(0, 4)) +
    theme_minimal(base_size = 11)
}

14 Adjusted CIF — Dose Discrepancy >= +25 mg (Carboplatin)

Standardized cumulative incidence functions for each competing-risk outcome, contrasting patients with dose_discrepancy >= 25 (red) against those with dose_discrepancy < 25 (blue). The Fine-Gray model is fit with the binary exposure plus the same adjuster set used in Section 7. Curves are adjusted (standardized) via g-computation: for every patient, predict CIF under exposure = 1 and under exposure = 0, then average the two CIF curves over the cohort. Y-axis is cumulative incidence, x-axis is days from baseline.

90-day death (Section 14.11) is modelled with Cox PH instead of Fine-Gray (no competing event), and the standardized cumulative-incidence-of-death curve is 1 − S(t|Xᵢ) averaged across the cohort.

adj_cif <- function(time_var, status_var, extra_pre = NULL, label) {
  base_adj <- c("age", "sex", "score", "ici")
  vars <- c("dose_discrep_ge25", base_adj, extra_pre)
  d <- dat %>%
    dplyr::select(all_of(c(time_var, status_var, vars))) %>%
    tidyr::drop_na()

  fmla <- as.formula(paste("~",
            paste(c("dose_discrep_ge25", base_adj, extra_pre), collapse = " + ")))
  X_full <- model.matrix(fmla, data = d)[, -1, drop = FALSE]

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- tryCatch(
    cmprsk::crr(ftime = time_num, fstatus = stat_num,
                cov1 = X_full, failcode = 1),
    error = function(e) NULL
  )
  if (is.null(fit)) return(.plot_failed(label))

  # subHR (95% CI) for the binary exposure, extracted from the same FG fit
  # dose_discrep_ge25 is the first column of cov1 (entered first in the formula)
  beta_e <- unname(fit$coef[1])
  se_e   <- sqrt(fit$var[1, 1])
  hr     <- exp(beta_e)
  lcl    <- exp(beta_e - 1.96 * se_e)
  ucl    <- exp(beta_e + 1.96 * se_e)
  pval   <- 2 * (1 - pnorm(abs(beta_e / se_e)))
  p_str  <- ifelse(pval < 0.001, "<0.001", sprintf("%.3f", pval))
  subtitle_str <- sprintf(
    "Discrep >= +25 vs <+25: subHR = %.2f (95%% CI %.2f-%.2f), p = %s",
    hr, lcl, ucl, p_str
  )

  X0 <- X_full; X0[, "dose_discrep_ge25"] <- 0
  X1 <- X_full; X1[, "dose_discrep_ge25"] <- 1

  pred0 <- predict(fit, X0)
  pred1 <- predict(fit, X1)

  times <- pred0[, 1]
  cif0  <- rowMeans(pred0[, -1, drop = FALSE])
  cif1  <- rowMeans(pred1[, -1, drop = FALSE])

  df <- bind_rows(
    tibble(time = times, cif = cif0, group = "Discrep < +25 mg"),
    tibble(time = times, cif = cif1, group = "Discrep >= +25 mg")
  )

  ymax <- max(df$cif, na.rm = TRUE)

  ggplot(df, aes(x = time, y = cif, color = group)) +
    geom_step(linewidth = 1) +
    scale_color_manual(values = c("Discrep < +25 mg"  = "#1F78B4",
                                  "Discrep >= +25 mg" = "#E31A1C")) +
    coord_cartesian(ylim = c(0, max(0.05, ymax * 1.05))) +
    labs(x = "Days from baseline",
         y = "Cumulative incidence",
         color = NULL,
         title = label,
         subtitle = subtitle_str) +
    theme_minimal(base_size = 11) +
    theme(legend.position = "top",
          plot.subtitle = element_text(size = 10, color = "grey25"))
}

adj_death_curve <- function(time_var = "time_to_death_90",
                            status_var = "death_90", label) {
  base_adj <- c("age", "sex", "score", "ici")
  vars <- c("dose_discrep_ge25", base_adj)
  d <- dat %>%
    dplyr::select(all_of(c(time_var, status_var, vars))) %>%
    tidyr::drop_na()

  rhs <- paste(c("dose_discrep_ge25", base_adj), collapse = " + ")
  fmla <- as.formula(sprintf("survival::Surv(%s, %s) ~ %s",
                             time_var, status_var, rhs))
  fit <- tryCatch(survival::coxph(fmla, data = d), error = function(e) NULL)
  if (is.null(fit)) return(.plot_failed(label))

  beta_e <- unname(coef(fit)["dose_discrep_ge25"])
  se_e   <- sqrt(vcov(fit)["dose_discrep_ge25", "dose_discrep_ge25"])
  hr     <- exp(beta_e)
  lcl    <- exp(beta_e - 1.96 * se_e)
  ucl    <- exp(beta_e + 1.96 * se_e)
  pval   <- 2 * (1 - pnorm(abs(beta_e / se_e)))
  p_str  <- ifelse(pval < 0.001, "<0.001", sprintf("%.3f", pval))
  subtitle_str <- sprintf(
    "Discrep >= +25 vs <+25: HR = %.2f (95%% CI %.2f-%.2f), p = %s   (Cox PH)",
    hr, lcl, ucl, p_str
  )

  d0 <- d; d0$dose_discrep_ge25 <- 0
  d1 <- d; d1$dose_discrep_ge25 <- 1

  sf0 <- survival::survfit(fit, newdata = d0)
  sf1 <- survival::survfit(fit, newdata = d1)

  s0 <- if (is.matrix(sf0$surv)) sf0$surv else as.matrix(sf0$surv)
  s1 <- if (is.matrix(sf1$surv)) sf1$surv else as.matrix(sf1$surv)

  cif0 <- 1 - rowMeans(s0)
  cif1 <- 1 - rowMeans(s1)

  df <- bind_rows(
    tibble(time = sf0$time, cif = cif0, group = "Discrep < +25 mg"),
    tibble(time = sf1$time, cif = cif1, group = "Discrep >= +25 mg")
  )

  ymax <- max(df$cif, na.rm = TRUE)

  ggplot(df, aes(x = time, y = cif, color = group)) +
    geom_step(linewidth = 1) +
    scale_color_manual(values = c("Discrep < +25 mg"  = "#1F78B4",
                                  "Discrep >= +25 mg" = "#E31A1C")) +
    coord_cartesian(ylim = c(0, max(0.05, ymax * 1.05))) +
    labs(x = "Days from baseline",
         y = "Cumulative incidence (1 - S)",
         color = NULL,
         title = label,
         subtitle = subtitle_str) +
    theme_minimal(base_size = 11) +
    theme(legend.position = "top",
          plot.subtitle = element_text(size = 10, color = "grey25"))
}
