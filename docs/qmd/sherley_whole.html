Event Counts by Outcome (Diagnostic)

For each outcome the table below shows the cohort N, events (status = 1 within 90 days), competing deaths (status = 2; not applicable to 90-day death which is modelled by Cox PH), censored, event rate, and events-per-variable (EPV) using the base adjuster set. As a rule of thumb, Fine-Gray fits need ≥ 5 × adj events to be stable; < 2 × adj almost always produces a singular design matrix, which is why the helper functions in this file fall back to a “Model could not be fit” placeholder where you see one.

.outcomes_tbl <- function(dat, cohort_label, n_adj = 16) {
  sumrow <- function(label, status_var, is_cox = FALSE) {
    s <- suppressWarnings(as.integer(as.character(dat[[status_var]])))
    n_total     <- sum(!is.na(s))
    n_event     <- sum(s == 1, na.rm = TRUE)
    n_compdeath <- if (is_cox) NA_integer_ else sum(s == 2, na.rm = TRUE)
    n_cens      <- sum(s == 0, na.rm = TRUE)
    epv         <- if (n_total > 0) n_event / n_adj else NA_real_
    fit_likely  <- if (is.na(epv))                "—"
                   else if (n_event >= 5 * n_adj) "✓ likely OK"
                   else if (n_event >= 2 * n_adj) "marginal"
                   else                           "✗ singular likely"
    tibble::tibble(
      Outcome            = label,
      N                  = n_total,
      Events             = n_event,
      `Competing deaths` = n_compdeath,
      Censored           = n_cens,
      `Event rate`       = if (n_total > 0) sprintf("%.1f%%", 100 * n_event / n_total) else "—",
      EPV                = if (!is.na(epv)) sprintf("%.1f", epv) else "—",
      `Fit reliable?`    = fit_likely
    )
  }
  dplyr::bind_rows(
    sumrow("Anemia Grade 2",                   "cmp_status_anemia_2"),
    sumrow("Anemia Grade 3",                   "cmp_status_anemia_3"),
    sumrow("Thrombocytopenia Grade 2",         "cmp_status_throm_2"),
    sumrow("Thrombocytopenia Grade 3",         "cmp_status_throm_3"),
    sumrow("AKI Grade 2",                      "cmp_status_aki_2"),
    sumrow("AKI (KDIGO)",                      "cmp_status_aki_kdigo"),
    sumrow("Composite Grade 2 AEs",            "cmp_status_grade_2_aes"),
    sumrow("Composite Grade 3 AEs",            "cmp_status_grade_3_aes"),
    sumrow("Overall Hospitalization",          "cmp_status_any_hosp"),
    sumrow("Platinum-Related Hospitalization", "cmp_status_side_effect"),
    sumrow("90-Day Death (Cox PH)",            "death_90", is_cox = TRUE)
  )
}

.adj_set <- function(pred_var, pre_labs, egfr_adj) {
  adj <- c("age","sex","score","ecog_score_cat2",
           "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
           "Paclitaxel","Pemetrexed","Gemcitabine","ici",
           "bmi","smoking","steroids","thyroid")
  # Drop platin_group when predictor requires dose data (carbo-only after drop_na)
  if (grepl("dose_discrep", pred_var)) adj <- setdiff(adj, "platin_group")
  if (egfr_adj == "cockcroft") adj <- c(adj, "cockcroft")
  if (egfr_adj == "ckd_epi")   adj <- c(adj, "ckd_epi_gfr_cre_cys_unindex")
  if (!is.null(pre_labs)) adj <- c(adj, pre_labs)
  adj
}

# fg_model: returns univariable + multivariable merged Fine-Gray table.
# If docx_name is supplied, also saves the table as a .docx in docx_dir.
fg_model <- function(time_var, status_var, pred_var, pre_labs = NULL,
                     egfr_adj = c("none", "cockcroft", "ckd_epi"),
                     docx_name = NULL) {
  egfr_adj <- match.arg(egfr_adj)
  adj <- .adj_set(pred_var, pre_labs, egfr_adj)
  all_vars <- c(pred_var, adj)

  # Multivariable
  fmla_mv <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(all_vars, collapse = " + ")
  ))
  tbl_mv <- tryCatch(
    tidycmprsk::crr(fmla_mv, failcode = 1, data = dat) %>%
      tbl_regression(exp = TRUE,
                     pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
      add_n(),
    error = function(e) NULL
  )

  # Univariable: one Fine-Gray fit per variable
  y_call <- str2lang(sprintf("Surv(%s, %s)", time_var, status_var))
  tbl_uv <- tryCatch(
    rlang::inject(
      tbl_uvregression(
        data = dat,
        method = tidycmprsk::crr,
        y = !!y_call,
        include = dplyr::all_of(all_vars),
        exponentiate = TRUE,
        pvalue_fun = function(x) style_pvalue(x, digits = 3),
        method.args = list(failcode = 1)
      )
    ),
    error = function(e) NULL
  )

  if (is.null(tbl_uv) && is.null(tbl_mv)) {
    return(knitr::asis_output(
      "> *Model could not be fit — Fine-Gray design matrix is singular (insufficient events or collinear covariates for this sub-cohort).*"
    ))
  }

  out <- if (is.null(tbl_mv)) {
    tbl_uv
  } else if (is.null(tbl_uv)) {
    tbl_mv
  } else {
    tbl_merge(
      tbls = list(tbl_uv, tbl_mv),
      tab_spanner = c("**Univariable**", "**Multivariable**")
    )
  }

  if (!is.null(docx_name)) save_tbl_docx(out, docx_name)
  out
}

# cox_model: same signature/output for the 90-day death analysis (Cox PH).
cox_model <- function(time_var, status_var, pred_var, pre_labs = NULL,
                      egfr_adj = c("none", "cockcroft", "ckd_epi"),
                      docx_name = NULL) {
  egfr_adj <- match.arg(egfr_adj)
  adj <- .adj_set(pred_var, pre_labs, egfr_adj)
  all_vars <- c(pred_var, adj)

  fmla_mv <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(all_vars, collapse = " + ")
  ))
  tbl_mv <- survival::coxph(fmla_mv, data = dat) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()

  y_call <- str2lang(sprintf("Surv(%s, %s)", time_var, status_var))
  tbl_uv <- rlang::inject(
    tbl_uvregression(
      data = dat,
      method = survival::coxph,
      y = !!y_call,
      include = dplyr::all_of(all_vars),
      exponentiate = TRUE,
      pvalue_fun = function(x) style_pvalue(x, digits = 3)
    )
  )

  out <- tbl_merge(
    tbls = list(tbl_uv, tbl_mv),
    tab_spanner = c("**Univariable**", "**Multivariable**")
  )

  if (!is.null(docx_name)) save_tbl_docx(out, docx_name)
  out
}

csh_model <- function(time_var, status_var, pred_var, pre_labs = NULL) {
  adj <- c("age","sex","score","ecog_score_cat2",
           "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
           "Paclitaxel","Pemetrexed","Gemcitabine","ici",
           "bmi","smoking","steroids","thyroid","cockcroft")
  if (grepl("dose_discrep", pred_var)) adj <- setdiff(adj, "platin_group")
  if (!is.null(pre_labs)) adj <- c(adj, pre_labs)
  # Cause-specific: treat death (status == 2) as censored
  d <- dat
  d$csh_status <- ifelse(d[[status_var]] == 1, 1L, 0L)
  fmla <- as.formula(paste0(
    "Surv(", time_var, ", csh_status) ~ ",
    paste(c(pred_var, adj), collapse = " + ")
  ))
  tryCatch(
    survival::coxph(fmla, data = d) %>%
      tbl_regression(exp = TRUE,
                     pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
      add_n(),
    error = function(e) .tbl_failed()
  )
}

NoteWhy anchor at the median (subHR = 1 at the reference)

Three reasons to anchor at the median:

  1. Interpretability: “A patient with eGFR ratio of 0.8 has ~1.5× the subHR compared to a typical patient (at the median).”
  2. Stability: The median sits in the densest part of the data, so estimates near it have the tightest confidence intervals. The CIs fan out as you move away — you can see this in the plots.
  3. No arbitrary choice: If we picked 1.0 arbitrarily, the plot would look almost identical here (because median ≈ 1), but in cohorts where the median is 0.9 or 1.1, picking 1.0 would either shift or distort the curve without changing the actual model.

plot_subhr_linear_ratio <- function(time_var, status_var, extra_adj = NULL,
                                    outcome_label, x_axis = c("ratio","percentile")) {
  x_axis <- match.arg(x_axis)
  base_adj <- c("age","sex","score","ecog_score_cat2","cockcroft",
                "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
                "Paclitaxel","Pemetrexed","Gemcitabine","ici",
                "bmi","smoking","steroids","thyroid")
  keep <- c("egfr_ratio", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(egfr_ratio = d$egfr_ratio, X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- tryCatch(
    cmprsk::crr(ftime = time_num, fstatus = stat_num,
                cov1 = X, failcode = 1),
    error = function(e) NULL
  )
  if (is.null(fit)) return(.plot_failed(outcome_label))

  beta   <- fit$coef[1]
  V_beta <- fit$var[1, 1]

  ref_x  <- median(d$egfr_ratio, na.rm = TRUE)
  grid_x <- seq(quantile(d$egfr_ratio, 0.025, na.rm = TRUE),
                quantile(d$egfr_ratio, 0.975, na.rm = TRUE),
                length.out = 100)

  log_hr <- beta * (grid_x - ref_x)
  se     <- sqrt((grid_x - ref_x)^2 * V_beta)

  if (x_axis == "percentile") {
    ecdf_fn      <- ecdf(d$egfr_ratio)
    grid_display <- ecdf_fn(grid_x) * 100
    ref_display  <- ecdf_fn(ref_x)  * 100
    xlab <- "Percentile of eGFR Ratio"
  } else {
    grid_display <- grid_x
    ref_display  <- ref_x
    xlab <- "CKD-EPI Cr-Cys / CG ratio"
  }

  df_plot <- tibble(
    x     = grid_display,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  p <- ggplot(df_plot, aes(x = x, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "darkorange") +
    geom_line(color = "darkorange", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_display, linetype = "dotted", color = "grey40") +
    labs(x = xlab, y = "subHR (95% CI)", title = outcome_label) +
    theme_minimal(base_size = 11)

  if (x_axis == "ratio") {
    p <- p + coord_cartesian(xlim = c(0.65, 1.45), ylim = c(0, 4))
  } else {
    p <- p + coord_cartesian(xlim = c(0.65, 1.45), ylim = c(0, 4))
  }
  p
}

plot_subhr_linear_disc <- function(time_var, status_var, extra_adj = NULL,
                                   outcome_label) {
  base_adj <- c("age","sex","score","ecog_score_cat2",
                "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
                "Paclitaxel","Pemetrexed","Gemcitabine","ici",
                "bmi","smoking","steroids","thyroid")
  keep <- c("egfr_discrepancy", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(egfr_discrepancy = d$egfr_discrepancy, X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- tryCatch(
    cmprsk::crr(ftime = time_num, fstatus = stat_num,
                cov1 = X, failcode = 1),
    error = function(e) NULL
  )
  if (is.null(fit)) return(.plot_failed(outcome_label))

  beta   <- fit$coef[1]
  V_beta <- fit$var[1, 1]
  se_b   <- sqrt(V_beta)

  # subHR summaries: per 1 mL/min, and per IQR (Q3 - Q1) of egfr_discrepancy
  q1    <- as.numeric(quantile(d$egfr_discrepancy, 0.25, na.rm = TRUE))
  q3    <- as.numeric(quantile(d$egfr_discrepancy, 0.75, na.rm = TRUE))
  iqr_w <- q3 - q1
  hr1   <- exp(beta);             hr1_l <- exp(beta - 1.96 * se_b);             hr1_u <- exp(beta + 1.96 * se_b)
  hri   <- exp(beta * iqr_w);     hri_l <- exp(beta * iqr_w - 1.96 * abs(iqr_w) * se_b); hri_u <- exp(beta * iqr_w + 1.96 * abs(iqr_w) * se_b)
  subtitle_str <- sprintf(
    "subHR per 1 mL/min: %.2f (%.2f-%.2f)  |  per IQR (Δ = %.1f): %.2f (%.2f-%.2f)  |  IQR: [%.1f, %.1f]",
    hr1, hr1_l, hr1_u, iqr_w, hri, hri_l, hri_u, q1, q3)

  ref_x  <- median(d$egfr_discrepancy, na.rm = TRUE)
  grid_x <- seq(quantile(d$egfr_discrepancy, 0.025, na.rm = TRUE),
                quantile(d$egfr_discrepancy, 0.975, na.rm = TRUE),
                length.out = 100)

  log_hr <- beta * (grid_x - ref_x)
  se     <- sqrt((grid_x - ref_x)^2 * V_beta)

  df_plot <- tibble(
    x     = grid_x,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  ggplot(df_plot, aes(x = x, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "darkorange") +
    geom_line(color = "darkorange", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_x, linetype = "dotted", color = "grey40") +
    geom_vline(xintercept = c(q1, q3), linetype = "dashed",
               color = "steelblue", alpha = 0.7) +
    labs(x = "eGFR Discrepancy (mL/min)",
         y = "subHR (95% CI)",
         title = outcome_label,
         subtitle = subtitle_str) +
    coord_cartesian(ylim = c(0, 4)) +
    theme_minimal(base_size = 11) +
    theme(plot.subtitle = element_text(size = 9, color = "grey25"))
}

plot_cg_vs_epi <- function(highlight_var, threshold, label) {

    filter(!is.na(cockcroft), !is.na(ckd_epi_gfr_cre_cys_unindex),

    mutate(highlight = .data[[highlight_var]] >= threshold)
